Web APIs Examples

Local Storage and Session Storage 
        // Storage API example
        localStorage.setItem('key', 'value');
        sessionStorage.setItem('key', 'value');

        const localStorageValue = localStorage.getItem('key');
        const sessionStorageValue = sessionStorage.getItem('key');

        document.getElementById('storageExample').textContent = `Local Storage: ${localStorageValue}\nSession Storage: ${sessionStorageValue}`;
Window History, Location, Navigator Objects 
        // Web APIs example
        // Accessing History Object
        const historyState = window.history.state;
        const historyLength = window.history.length;

        // Accessing Location Object
        const currentURL = window.location.href;
        const hostname = window.location.hostname;
        const pathname = window.location.pathname;

        // Accessing Navigator Object
        const userAgent = window.navigator.userAgent;
        const platform = window.navigator.platform;

        // Displaying information in webAPIsExample
        const webAPIsInfo = `History State: ${historyState}\nHistory Length: ${historyLength}\nCurrent URL: ${currentURL}\nHostname: ${hostname}\nPathname: ${pathname}\nUser Agent: ${userAgent}\nPlatform: ${platform}`;
        document.getElementById('webAPIsExample').textContent = webAPIsInfo;
Fetch API 
        // Fetch API example
        fetch('https://jsonplaceholder.typicode.com/posts/1')
            .then(response => {
                if (!response.ok) {
                    throw new Error('Network response was not ok');
                }
                return response.json();
            })
            .then(data => {
                const formattedData = JSON.stringify(data, null, 4); // Format JSON for better readability
                document.getElementById('fetchExample').textContent = formattedData;
            })
            .catch(error => {
                document.getElementById('fetchExample').textContent = `Fetch API error: ${error.message}`;
            });
