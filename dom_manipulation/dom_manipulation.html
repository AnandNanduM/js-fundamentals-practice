    const paragraphElements = document.getElementsByClassName('content');

    console.log(paragraphElements[0].textContent); // Output: This is a paragraph.
    
    // Selecting elements by tag name

Modifying the element prototype 
 
        
// Define a custom method to capitalize the text content of an element
HTMLElement.prototype.capitalizeText = function() {
    // Ensure the element has text content
    if (this.textContent) {
        // Capitalize the text content
        this.textContent = this.textContent.toUpperCase();
    }
};

// Usage example:
// Select an element by its ID
const headingElement = document.getElementById('heading');
// Call the custom method to capitalize the text content
headingElement.capitalizeText();
Dynamic Content Example 
        // Create a new paragraph element
        const paragraph = document.createElement('p');
        paragraph.textContent = 'This is a dynamically created paragraph.';

        // Create a new button element
        const button = document.createElement('button');
        button.textContent = 'Click me';

        // Add a click event listener to the button
        button.addEventListener('click', function() {
            alert('Button clicked!');
        });

        // Select the container div
        const container = document.getElementById('container');

        // Append the paragraph and button elements to the container div
        container.appendChild(paragraph);
        container.appendChild(button);
