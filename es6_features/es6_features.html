ES6+ Features Example

Template Literals 
        // Template literals
        const name = 'John';
        const greeting = `Hello, ${name}!`;
        document.getElementById('templateLiterals').textContent = greeting;
Destructuring Assignments 
        // Destructuring assignments
        const person = { firstName: 'Alice', lastName: 'Smith' };
        const { firstName, lastName } = person;
        document.getElementById('destructuringAssignments').textContent = `${firstName} ${lastName}`;
Rest and Spread Operator 
        // Rest and spread operator
        const numbers = [1, 2, 3, 4, 5];
        const [first, second, ...rest] = numbers;
        document.getElementById('restAndSpreadOperator').textContent = `Rest operator: ${first} ${second}\nSpread operator: ${rest}`;
Arrow Functions 
        // Arrow functions
        const add = (a, b) => a + b;
        document.getElementById('arrowFunctions').textContent = `Arrow functions: ${add(3, 4)}`;
Classes and Inheritance 
        // Classes and inheritance
        class Person {
            constructor(name) {
                this.name = name;
            }

            greet() {
                return `Hello, ${this.name}!`;
            }
        }

        class Student extends Person {
            constructor(name, grade) {
                super(name);
                this.grade = grade;
            }

            displayInfo() {
                return `Name: ${this.name}, Grade: ${this.grade}`;
            }
        }

        const student = new Student('Emma', 'A');
        document.getElementById('classesAndInheritance').textContent = `${student.greet()}\n${student.displayInfo()}`;
Default Parameters 
        // Default parameters
        function greet(name = 'Guest') {
            return `Hello, ${name}!`;
        }

        document.getElementById('defaultParameters').textContent = `Default parameters: ${greet()}`;
